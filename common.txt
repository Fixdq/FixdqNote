删除软件

# 删除软件及其配置文件
sudo apt-get --purge remove <package>

# 删除没用的依赖包
sudo apt-get autoremove <package>

# 此时dpkg的列表中有“rc”状态的软件包，可以执行如下命令做最后清理：
dpkg -l |grep ^rc|awk '{print $2}' |sudo xargs dpkg -P
